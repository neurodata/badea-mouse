correlations, labels = load_volume_corr()

## Plot to make sure nothing is wrong
fig, ax = plt.subplots(1, 3, figsize=(6, 2), dpi=300)


for idx, label in enumerate(labels):
    sns.heatmap(
        correlations[idx],
        ax=ax[idx],
        vmin=-1,
        vmax=1,
        cbar=False,
        cmap="RdBu_r",
        xticklabels=False,
        yticklabels=False,
